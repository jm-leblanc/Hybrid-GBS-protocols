#!/usr/bin/env nextflow
nextflow.enable.dsl = 2

// ------------------------------------------------------------
// This script runs ENTROPY for 3 replicates of each value of k (specified below)'
// Input: An MPGL file created from a filtered VCF, plus the initial q values for each k, created during VCF --> MPGL conversion ("qk${k}inds_FIXED.txt" files)
// Output: HDF5 files for 3 reps for each value of k
// Author: Jenna LeBlanc
// Created: August 2026
// Last Updated: March 10, 2026
// This is the most recent and correct version of the script

// Note for users:
// To actually run this script, copy its contents into a file called
// "main.nf" and run:
//    nextflow run main.nf
// ------------------------------------------------------------

// Parameters
params.k_values   = "2,3,4,5,6,7,8,9"
params.replicates = 3
params.outdir     = "results"

// Location of fixed input files
params.inputdir   = "/home/lebla179/scratch/entropy_k_2-9/MPGL_conversion/output_fixed"

process ENTROPY {
    publishDir "${params.outdir}", mode: 'copy'

    input:
    tuple val(k), val(rep)

    output:
    path "fescue_k${k}_rep${rep}.hdf5"

    script:
    """
    /home/lebla179/scratch/entropy/software/entropy \\
        -i ${params.inputdir}/fescue_biallelic_snps_only.recode_FIXED.mpgl \\
        -n 2 \\
        -k ${k} \\
        -q ${params.inputdir}/qk${k}inds_FIXED.txt \\
        -l 60000 \\
        -b 20000 \\
        -t 20 \\
        -r \$RANDOM \\
        -s 20 \\
        -e 0.01 \\
        -o fescue_k${k}_rep${rep}.hdf5
    """
}

workflow {
    // All (K, replicate) combinations
    Channel
        .from(params.k_values.split(','))
        .map { it.toInteger() }
        .combine(Channel.from(1..params.replicates))
        .set { combinations }

    ENTROPY(combinations)
}
