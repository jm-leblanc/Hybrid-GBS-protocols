MAIN.NF
#!/usr/bin/env nextflow

nextflow.enable.dsl = 2

process VCF2PLOIDY {
    tag "Estimating ploidy from ${vcf_file}"
    publishDir "${params.outdir}", mode: 'copy'

    input:
    path vcf_file

    output:
    path "ploidy_results.rds", emit: results
    path "ploidy_summary.csv", emit: summary
    path "ploidy_analysis.log", emit: log

    script:
    def props_vector = params.props.split(',').collect { it.trim() }.join(', ')
    """
    #!/usr/bin/env Rscript
    
    # Load required libraries
    library(vcf2ploidy)
    
    # Start logging
    cat("=== VCF2PLOIDY Analysis Started ===\\n", file="ploidy_analysis.log")
    cat("Time:", as.character(Sys.time()), "\\n", file="ploidy_analysis.log", append=TRUE)
    cat("Input VCF:", "${vcf_file}", "\\n", file="ploidy_analysis.log", append=TRUE)
    cat("Expected runtime: Several hours for MCMC...\\n", file="ploidy_analysis.log", append=TRUE)
    
    # Log parameters
    cat("\\n=== Parameters ===\\n", file="ploidy_analysis.log", append=TRUE)
    cat("Remove double hets:", ${params.remove_double_hets}, "\\n", file="ploidy_analysis.log", append=TRUE)
    cat("Number of classes:", ${params.nclasses}, "\\n", file="ploidy_analysis.log", append=TRUE)
    cat("MCMC chains:", ${params.mcmc_nchain}, "\\n", file="ploidy_analysis.log", append=TRUE)
    cat("MCMC steps:", ${params.mcmc_steps}, "\\n", file="ploidy_analysis.log", append=TRUE)
    cat("MCMC burnin:", ${params.mcmc_burnin}, "\\n", file="ploidy_analysis.log", append=TRUE)
    cat("MCMC thin:", ${params.mcmc_thin}, "\\n", file="ploidy_analysis.log", append=TRUE)
    
    # Run vcf2ploidy - one function does it all!
    cat("\\n=== Running vcf2ploidy (this will take hours) ===\\n", file="ploidy_analysis.log", append=TRUE)
    
    results <- vcf2ploidy(
        filename = "${vcf_file}",
        remove_double_hets = ${params.remove_double_hets ? 'TRUE' : 'FALSE'},
        props = c(${props_vector}),
        mcmc.nchain = ${params.mcmc_nchain},
        mcmc.steps = ${params.mcmc_steps},
        mcmc.burnin = ${params.mcmc_burnin},
        mcmc.thin = ${params.mcmc_thin},
        train = FALSE,
        nclasses = ${params.nclasses},
        pcs = 1:2
    )
    
    cat("vcf2ploidy completed successfully!\\n", file="ploidy_analysis.log", append=TRUE)
    
    # Save results
    saveRDS(results, "ploidy_2class_results.rds")
    
    # Create summary
    if (!is.null(results\$pp)) {
        pp <- results\$pp
        
        # Extract sample IDs and probabilities
        if (ncol(pp) > ${params.nclasses}) {
            sample_ids <- pp[,1]
            prob_matrix <- apply(pp[,2:ncol(pp)], 2, as.numeric)
        } else {
            sample_ids <- paste0("Sample_", 1:nrow(pp))
            prob_matrix <- apply(pp, 2, as.numeric)
        }
        
        # Get assignments
        class_assignments <- apply(prob_matrix, 1, which.max)
        max_probabilities <- apply(prob_matrix, 1, max)
        
        # Create summary
        summary_df <- data.frame(
            Sample_ID = sample_ids,
            Predicted_Ploidy_Class = class_assignments,
            Max_Probability = round(max_probabilities, 3)
        )
        
        # Add class probabilities
        for (i in 1:ncol(prob_matrix)) {
            summary_df[[paste0("Class_", i, "_Prob")]] <- round(prob_matrix[,i], 3)
        }
        
        write.csv(summary_df, "ploidy_summary.csv", row.names = FALSE)
        
        # Log summary stats
        cat("\\n=== Results Summary ===\\n", file="ploidy_analysis.log", append=TRUE)
        cat("Total samples:", nrow(summary_df), "\\n", file="ploidy_analysis.log", append=TRUE)
        for (i in 1:${params.nclasses}) {
            count <- sum(class_assignments == i)
            cat("Class", i, "samples:", count, "\\n", file="ploidy_analysis.log", append=TRUE)
        }
    }
    
    cat("\\nAnalysis completed at:", as.character(Sys.time()), "\\n", file="ploidy_analysis.log", append=TRUE)
    """
}

workflow {
    vcf_ch = Channel.fromPath(params.vcf_file, checkIfExists: true)
    VCF2PLOIDY(vcf_ch)
}
