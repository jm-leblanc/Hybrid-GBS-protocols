library(vcfR)
library(dplyr)
vcf <- read.vcfR("fescue_biallelic_snps_only.recode.vcf") # Filtered VCF, further filtered to remove multiallelic sites and indels

sample_ids <- colnames(vcf@gt)[-1]
sample_ids

ploidy_vector <- rep(NA, length(sample_ids))

ploidy_vector

names(ploidy_vector) <- sample_ids

ploidy_vector


# Set known ploidy for training samples
ploidy_vector[c("Fescue34", "Fescue35", "Fescue36", "Fescue37", "Fescue38")] <- 8

ploidy_vector[c("Fescue69", "Fescue70", "Fescue71", "Fescue72", "Fescue73")] <- 4

# Training indices: samples with known ploidy
training_indices <- which(!is.na(ploidy_vector))


ploidy_vector
training_indices

# Save sample IDs as a plain vector (one per line)
writeLines(names(ploidy_vector), "sample_ids.txt")

# Save ploidy vector as a two-column tab-separated file: sample_id and ploidy
write.table(
  data.frame(sample_id = names(ploidy_vector), ploidy = ploidy_vector),
  file = "ploidy_vector.tsv",
  sep = "\t",
  quote = FALSE,
  row.names = FALSE,
  col.names = TRUE
)
